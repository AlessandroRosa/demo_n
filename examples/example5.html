Example 5: fading images
var _demon = new demo_n( true );
    _demon.add( { action : "start" } );

    _demon.add( { pre_fn : function(){ $( "#test_img" ).prop( "src", "pix/pix.1.jpg" ) ; },
                  ctrl_id : "test_img", time : 2, time_unit : "s",
                  action : "nothing", set_value : ""
                } );

    _demon.add( { ctrl_id : "test_img", time : 2, time_unit : "s",
                  action : "fadein", set_value : "slow"
                } );

    _demon.add( { ctrl_id : "test_img", time : 2.5, time_unit : "s",
                  action : "fadeout", set_value : "slow",
                  post_fn : function(){ setTimeout( function() { $( "#test_img" ).prop( "src", "pix/pix.2.jpg" ) ; }, 1500 ) ; }
                } );

    _demon.add( { ctrl_id : "test_img", time : 2, time_unit : "s",
                  action : "fadein", set_value : "slow"
                } );

    _demon.add( { ctrl_id : "test_img", time : 2.5, time_unit : "s",
                  action : "fadeout", set_value : "slow",
                  post_fn : function(){ setTimeout( function() { $( "#test_img" ).prop( "src", "pix/pix.3.jpg" ) ; }, 1500 ) ; }
                } );

    _demon.add( { ctrl_id : "test_img", time : 2, time_unit : "s",
                  action : "fadein", set_value : "slow"
                } );

    _demon.add( { ctrl_id : "test_img", time : 2, time_unit : "s",
                  action : "fadeout", set_value : "slow"
                } );

    _demon.add( { action : "end" } );
