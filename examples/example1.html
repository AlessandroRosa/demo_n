Example 1: clicking on some controls
var _demon = new demo_n(true);
    _demon.add( { action : "start" } );

    _demon.add( { ctrl_id : "radio1", time : 1, time_unit : "s",
                  action : "click", showlabel : true, desclabel : "First, click radio 1"
                } );

    _demon.add( { ctrl_id : "radio2", time : 1, time_unit : "s",
                  action : "click", showlabel : true, desclabel : "Then, click radio 2"
                } );

    _demon.add( { ctrl_id : "radio3", time : 1, time_unit : "s",
                  action : "click", showlabel : true, desclabel : "Finally, click radio 3"
                } );

    _demon.add( { ctrl_id : "mychkbox", time : 1, time_unit : "s",
                  action : "nothing", set_value : -1, showlabel : true, desclabel : "I'm pointing to this checkbox" } );
    _demon.add( { ctrl_id : "mychkbox", time : 1, time_unit : "s",
                  action : "click", set_value : -1, showlabel : true, desclabel : "I selected the checkbox" } );

    _demon.add( { action : "end" } );
    _demon.run();
After initializing the DemoN object, we added the three click-actions over some radio buttons.
The time interval between each action is 1 second.
Finally we move to a checkbox: we focus on it and then we click.
